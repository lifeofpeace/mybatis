16  package org.apache.ibatis.parsing;
17  
18  import static org.junit.Assert.assertEquals;
19  import org.junit.Test;
20  
21  import java.util.HashMap;
22  import java.util.Map;
23  
24  public class GenericTokenParserTest {
25  
26    public static class VariableTokenHandler implements TokenHandler {
27      private Map<String, String> variables = new HashMap<String, String>();
28  
29      public VariableTokenHandler(Map<String, String> variables) {
30        this.variables = variables;
31      }
32  
33      public String handleToken(String content) {
34        return variables.get(content);
35      }
36    }
37  
38    @Test
39    public void shouldDemonstrateGenericTokenReplacement() {
40      GenericTokenParser parser = new GenericTokenParser("${", "}", new VariableTokenHandler(new HashMap<String, String>() {
41        {
42          put("first_name", "James");
43          put("initial", "T");
44          put("last_name", "Kirk");
45          put("", "");
46        }
47      }));
48  
49      assertEquals("James T Kirk reporting.", parser.parse("${first_name} ${initial} ${last_name} reporting."));
50      assertEquals("Hello captain James T Kirk", parser.parse("Hello captain ${first_name} ${initial} ${last_name}"));
51      assertEquals("James T Kirk", parser.parse("${first_name} ${initial} ${last_name}"));
52      assertEquals("JamesTKirk", parser.parse("${first_name}${initial}${last_name}"));
53      assertEquals("{}JamesTKirk", parser.parse("{}${first_name}${initial}${last_name}"));
54      assertEquals("}JamesTKirk", parser.parse("}${first_name}${initial}${last_name}"));
55  
56      assertEquals("}James{{T}}Kirk", parser.parse("}${first_name}{{${initial}}}${last_name}"));
57      assertEquals("}James}T{Kirk", parser.parse("}${first_name}}${initial}{${last_name}"));
58      assertEquals("}James}T{Kirk", parser.parse("}${first_name}}${initial}{${last_name}"));
59      assertEquals("}James}T{Kirk{{}}", parser.parse("}${first_name}}${initial}{${last_name}{{}}"));
60      assertEquals("}James}T{Kirk{{}}", parser.parse("}${first_name}}${initial}{${last_name}{{}}${}"));
61  
62      assertEquals("{$$something}JamesTKirk", parser.parse("{$$something}${first_name}${initial}${last_name}"));
63      assertEquals("${", parser.parse("${"));
64      assertEquals("}", parser.parse("}"));
65      assertEquals("Hello ${ this is a test.", parser.parse("Hello ${ this is a test."));
66      assertEquals("Hello } this is a test.", parser.parse("Hello } this is a test."));
67      assertEquals("Hello } ${ this is a test.", parser.parse("Hello } ${ this is a test."));
68    }
69  
70    @Test
71    public void shallNotInterpolateSkippedVaiables() {
72      GenericTokenParser parser = new GenericTokenParser("${", "}", new VariableTokenHandler(new HashMap<String, String>()));
73  
74      assertEquals("${skipped} variable", parser.parse("\\${skipped} variable"));
75      assertEquals("This is a ${skipped} variable", parser.parse("This is a \\${skipped} variable"));
76      assertEquals("null ${skipped} variable", parser.parse("${skipped} \\${skipped} variable"));
77      assertEquals("The null is ${skipped} variable", parser.parse("The ${skipped} is \\${skipped} variable"));
78    }
79  
80    @Test(timeout = 1000)
81    public void shouldParseFastOnJdk7u6() {
82      // issue #760
83      GenericTokenParser parser = new GenericTokenParser("${", "}", new VariableTokenHandler(new HashMap<String, String>() {
84        {
85          put("first_name", "James");
86          put("initial", "T");
87          put("last_name", "Kirk");
88          put("", "");
89        }
90      }));
91  
92      StringBuilder input = new StringBuilder();
93      for (int i = 0; i < 10000; i++) {
94        input.append("${first_name} ${initial} ${last_name} reporting. ");
95      }
96      StringBuilder expected = new StringBuilder();
97      for (int i = 0; i < 10000; i++) {
98        expected.append("James T Kirk reporting. ");
99      }
100     assertEquals(expected.toString(), parser.parse(input.toString()));
101   }

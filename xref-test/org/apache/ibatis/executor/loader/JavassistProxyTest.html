16  package org.apache.ibatis.executor.loader;
17  
18  import static org.junit.Assert.assertEquals;
19  import static org.junit.Assert.assertFalse;
20  import static org.junit.Assert.assertTrue;
21  
22  import java.io.Serializable;
23  import java.util.ArrayList;
24  import java.util.HashMap;
25  
26  import javassist.util.proxy.Proxy;
27  
28  import org.apache.ibatis.domain.blog.Author;
29  import org.apache.ibatis.executor.ExecutorException;
30  import org.apache.ibatis.reflection.factory.DefaultObjectFactory;
31  import org.apache.ibatis.session.Configuration;
32  import org.junit.Test;
33  
34  public class JavassistProxyTest extends SerializableProxyTest {
35  
36    public JavassistProxyTest() {
37      proxyFactory = new JavassistProxyFactory();
38    }
39  
40    @Test
41    public void shouldCreateAProxyForAPartiallyLoadedBean() throws Exception {
42      ResultLoaderMap loader = new ResultLoaderMap();
43      loader.addLoader("id", null, null);
44      Object proxy = proxyFactory.createProxy(author, loader, new Configuration(), new DefaultObjectFactory(), new ArrayList<Class<?>>(), new ArrayList<Object>());
45      Author author2 = (Author) deserialize(serialize((Serializable) proxy));
46      assertTrue(author2 instanceof Proxy);
47    }
48  
49    @Test(expected = ExecutorException.class)
50    public void shouldFailCallingAnUnloadedProperty() throws Exception {
51      // yes, it must go in uppercase
52      HashMap<String, ResultLoaderMap.LoadPair> unloadedProperties = new HashMap<String, ResultLoaderMap.LoadPair> ();
53      unloadedProperties.put("ID", null);
54      Author author2 = (Author) ((JavassistProxyFactory)proxyFactory).createDeserializationProxy(author, unloadedProperties, new DefaultObjectFactory(), new ArrayList<Class<?>>(), new ArrayList<Object>());
55      author2.getId();
56    }
57  
58    @Test
59    public void shouldLetCallALoadedProperty() throws Exception {
60      Author author2 = (Author) ((JavassistProxyFactory)proxyFactory).createDeserializationProxy(author, new HashMap<String, ResultLoaderMap.LoadPair>(), new DefaultObjectFactory(), new ArrayList<Class<?>>(), new ArrayList<Object>());
61      assertEquals(999, author2.getId());
62    }
63  
64    @Test
65    public void shouldSerizalizeADeserlizaliedProxy() throws Exception {
66      Object proxy = ((JavassistProxyFactory)proxyFactory).createDeserializationProxy(author, new HashMap<String, ResultLoaderMap.LoadPair> (), new DefaultObjectFactory(), new ArrayList<Class<?>>(), new ArrayList<Object>());
67      Author author2 = (Author) deserialize(serialize((Serializable) proxy));
68      assertEquals(author, author2);
69      assertFalse(author.getClass().equals(author2.getClass()));
70    }
71  
72  }

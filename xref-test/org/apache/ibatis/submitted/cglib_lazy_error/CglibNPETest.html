35    @BeforeClass
36    public static void initDatabase() throws Exception {
37      Connection conn = null;
38  
39      try {
40        Class.forName("org.hsqldb.jdbcDriver");
41        conn = DriverManager.getConnection("jdbc:hsqldb:mem:cglib_lazy_eager_no_error", "sa",
42            "");
43  
44        Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/cglib_lazy_error/CreateDB.sql");
45  
46        ScriptRunner runner = new ScriptRunner(conn);
47        runner.setLogWriter(null);
48        runner.setErrorLogWriter(null);
49        runner.runScript(reader);
50        conn.commit();
51        reader.close();
52  
53        reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/cglib_lazy_error/ibatisConfig.xml");
54        sqlSessionFactory = new SqlSessionFactoryBuilder().build(reader);
55        reader.close();
56      } finally {
57        if (conn != null) {
58          conn.close();
59        }
60      }
61    }

63    @Test
64    public void testNoParent() {
65      SqlSession sqlSession = sqlSessionFactory.openSession();
66      PersonMapper personMapper = sqlSession.getMapper(PersonMapper.class);
67      Person person = personMapper.selectById(1);
68      Assert.assertNotNull("Persons must not be null", person);
69      Person parent = person.getParent();
70      Assert.assertNull("Parent must be null", parent);
71      sqlSession.close();
72    }

74    @Test
75    public void testAncestorSelf() {

77      PersonMapper personMapper = sqlSession.getMapper(PersonMapper.class);
78      Person person = personMapper.selectById(1);
79      Assert.assertNotNull("Persons must not be null", person);
80      Person ancestor = person.getAncestor();

85    @Test
86    public void testGrandParent() {
87      SqlSession sqlSession = sqlSessionFactory.openSession();
88      PersonMapper personMapper = sqlSession.getMapper(PersonMapper.class);
89      Person expectedParent = personMapper.selectById(2);
90      Person expectedGrandParent = personMapper.selectById(1);
91      Person person = personMapper.selectById(3);
92      Assert.assertNotNull("Persons must not be null", person);
93      Assert.assertEquals("Parent must be John Smith", expectedParent, person.getParent());
94      Assert.assertEquals("Parent must be John Smith sr.", expectedGrandParent, person.getParent().getParent());
95      sqlSession.close();
96    }

109   @Test
110   public void testAncestorAfterQueryingParents() {
111     SqlSession sqlSession = sqlSessionFactory.openSession();
112     PersonMapper personMapper = sqlSession.getMapper(PersonMapper.class);
113     Person expectedAncestor = personMapper.selectById(1);
114     Person person = personMapper.selectById(3);
115     // Load ancestor indirectly.
116     Assert.assertNotNull("Persons must not be null", person);
117     Assert.assertNotNull("Parent must not be null", person.getParent());
118     Assert.assertNotNull("Grandparent must not be null", person.getParent().getParent());
119     Assert.assertEquals("Ancestor must be John Smith sr.", expectedAncestor, person.getAncestor());
120     sqlSession.close();
121   }

123   @Test
124   public void testInsertBetweenTwoSelects() {
125     SqlSession sqlSession = sqlSessionFactory.openSession();
126     try {
127       PersonMapper personMapper = sqlSession.getMapper(PersonMapper.class);
128       Person selected1 = personMapper.selectById(1);
129       Person selected2 = personMapper.selectById(2);
130       Person selected3 = personMapper.selectById(3);
131       selected1.setId(4L);
132       int rows = personMapper.insertPerson(selected1);
133       Assert.assertEquals(1, rows);
134       selected1 = personMapper.selectById(1);
135       selected2 = personMapper.selectById(2);
136       selected3 = personMapper.selectById(3);
137       Person selected4 = personMapper.selectById(4);
138       Assert.assertEquals(1, selected1.getId().longValue());
139       Assert.assertEquals(2, selected2.getId().longValue());
140       Assert.assertEquals(3, selected3.getId().longValue());
141       Assert.assertEquals(4, selected4.getId().longValue());
142 
143     } finally {
144       sqlSession.close();
145     }
146   }

16  package org.apache.ibatis.submitted.count;
17  
18  import static org.junit.Assert.assertEquals;
19  
20  import java.io.Reader;
21  import java.sql.Connection;
22  import java.sql.DriverManager;
23  
24  import org.apache.ibatis.io.Resources;
25  import org.apache.ibatis.jdbc.ScriptRunner;
26  import org.apache.ibatis.session.SqlSession;
27  import org.apache.ibatis.session.SqlSessionFactory;
28  import org.apache.ibatis.session.SqlSessionFactoryBuilder;
29  import org.junit.BeforeClass;
30  import org.junit.Test;
31  
32  public class CountTest {
33  
34    protected static SqlSessionFactory sqlSessionFactory;
35  
36    @BeforeClass
37    public static void setUp() throws Exception {
38      Connection conn = null;
39  
40      try {
41        Class.forName("org.hsqldb.jdbcDriver");
42        conn = DriverManager.getConnection("jdbc:hsqldb:mem:gname", "sa",
43            "");
44  
45        Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/count/CreateDB.sql");
46  
47        ScriptRunner runner = new ScriptRunner(conn);
48        runner.setLogWriter(null);
49        runner.setErrorLogWriter(null);
50        runner.runScript(reader);
51        conn.commit();
52        reader.close();
53  
54        reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/count/MapperConfig.xml");
55        sqlSessionFactory = new SqlSessionFactoryBuilder().build(reader);
56        reader.close();
57      } finally {
58        if (conn != null) {
59          conn.close();
60        }
61      }
62    }
63  
64    @Test
65    public void testCount() {
66      SqlSession sqlSession = sqlSessionFactory.openSession();
67      try {
68        CountMapper mapper = sqlSession.getMapper(CountMapper.class);
69        int answer = mapper.count();
70        assertEquals(6, answer);
71      } finally {
72        sqlSession.close();
73      }
74    }
75  }

16  package org.apache.ibatis.type;
17  
18  import static org.junit.Assert.assertEquals;
19  import static org.mockito.Mockito.verify;
20  import static org.mockito.Mockito.when;
21  
22  import java.sql.Timestamp;
23  import java.util.Date;
24  
25  import org.junit.Test;
26  
27  public class DateTypeHandlerTest extends BaseTypeHandlerTest {
28  
29    private static final TypeHandler<Date> TYPE_HANDLER = new DateTypeHandler();
30    private static final Date DATE = new Date();
31    private static final Timestamp TIMESTAMP = new Timestamp(DATE.getTime());
32  
33    @Test
34    public void shouldSetParameter() throws Exception {
35      TYPE_HANDLER.setParameter(ps, 1, DATE, null);
36      verify(ps).setTimestamp(1, new java.sql.Timestamp(DATE.getTime()));
37    }
38  
39    @Test
40    public void shouldGetResultFromResultSet() throws Exception {
41      when(rs.getTimestamp("column")).thenReturn(TIMESTAMP);
42      when(rs.wasNull()).thenReturn(false);
43      assertEquals(DATE, TYPE_HANDLER.getResult(rs, "column"));
44    }
45  
46    @Test
47    public void shouldGetResultFromCallableStatement() throws Exception {
48      when(cs.getTimestamp(1)).thenReturn(TIMESTAMP);
49      when(cs.wasNull()).thenReturn(false);
50      assertEquals(DATE, TYPE_HANDLER.getResult(cs, 1));
51    }
52  
53  }
